<script>

const HALF_LIFE_HOURS = 120;

const k = Math.log(2) / HALF_LIFE_HOURS;


let injections = JSON.parse(localStorage.getItem("inj") || "[]");


function save(){

  localStorage.setItem("inj", JSON.stringify(injections));

}


function addWeight(){}


function addInjection(){

  const val = parseFloat(document.getElementById("doseInput").value);

  if(!val) return;


  injections.push({time:Date.now(), dose:val});

  save();

  render();

}


function calc(){

  const now = Date.now();

  let total = 0;


  injections.forEach(i=>{

    const dt = (now - i.time) / (1000*60*60);

    if(dt >= 0){

      total += i.dose * Math.exp(-k * dt);

    }

  });


  return total.toFixed(2);

}


function render(){

  document.getElementById("level").innerText =

    "Current Level: " + calc();

}


render();

</script>
