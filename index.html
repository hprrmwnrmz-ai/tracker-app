<script>


const HALF_LIFE_HOURS = 120;

const k = Math.log(2) / HALF_LIFE_HOURS;


let weights = load("weights", []);

let injections = load("injections", []);

let view = "main";


function load(key, fallback){

  try {

    return JSON.parse(localStorage.getItem(key)) || fallback;

  } catch {

    return fallback;

  }

}


function save(key, data){

  localStorage.setItem(key, JSON.stringify(data));

}


function addWeight(){

  const val = parseFloat(document.getElementById("weightInput").value);

  if(!val) return;


  weights.push({

    time: Date.now(),

    weight: val

  });


  save("weights", weights);

  document.getElementById("weightInput").value = "";

  render();

}


function addInjection(){

  const val = parseFloat(document.getElementById("doseInput").value);

  if(!val) return;


  injections.push({

    time: Date.now(),

    dose: val

  });


  save("injections", injections);

  document.getElementById("doseInput").value = "";

  render();

}


function calcLevel(){

  const now = Date.now();

  let total = 0;


  injections.forEach(i=>{

    const dt = (now - i.time) / (1000*60*60);

    if(dt >= 0){

      total += i.dose * Math.exp(-k * dt);

    }

  });


  return total.toFixed(2);

}


function toggleView(){

  view = view === "main" ? "level" : "main";

  document.getElementById("mainView").style.display = view==="main"?"block":"none";

  document.getElementById("levelView").style.display = view==="level"?"block":"none";

  render();

}


function render(){


  // summary

  document.getElementById("summary").innerHTML =

    weights.length + " weights • " + injections.length + " injections";


  // entries

  const list = document.getElementById("entries");

  list.innerHTML = "";


  weights.slice().reverse().forEach(w=>{

    const div = document.createElement("div");

    div.className = "card";

    div.innerHTML =

      new Date(w.time).toLocaleString() + "<br>" +

      w.weight + " lbs";

    list.appendChild(div);

  });


  // level

  document.getElementById("levelValue").innerText = calcLevel();

}


render();


</script>
